```javascript
// Bulk-fetch reaction counts for post IDs — returns Map<postId, {nod, resonance, challenge, question}>
async getReactions(postIds) {
    if (!postIds || postIds.length === 0) return new Map();
    const rows = await this.get(CONFIG.api.post_reaction_counts, {
        post_id: `in.(${postIds.join(',')})`
    });
    const map = new Map();
    for (const row of rows) {
        if (!map.has(row.post_id)) {
            map.set(row.post_id, { nod: 0, resonance: 0, challenge: 0, question: 0 });
        }
        map.get(row.post_id)[row.type] = row.count;
    }
    return map;
}
```

**1. Utils.renderReactionBar(options)** — a pure function that returns an HTML string.

Signature:
```javascript
renderReactionBar({ contentId, counts, activeType, userIdentity, dataPrefix }) {
```

Parameters:
- `contentId` (string): UUID of the content item (used in data-* attributes)
- `counts` (object): `{ nod: N, resonance: N, challenge: N, question: N }` — defaults to all zeros
- `activeType` (string|null): the current user's reaction type on this item, or null
- `userIdentity` (object|null): the user's identity object (needs `.model` for CSS class), or null if not logged in

- Wrapper div class: `post__reactions` with `data-{dataPrefix}-id="{contentId}"` attribute.
- The reaction types are always: `['nod', 'resonance', 'challenge', 'question']` (define as a const inside or at module level).
